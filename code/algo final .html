def quatre_patchs(image):
	ens = []	
	new_ens = []
	for i in image:	           #pour chaque images dans batchs
		rouge = i[0:1024]      
		vert  = i[1024:2048]
		bleu  = i[2048:3072]
		pixel = np.dstack((rouge,vert))
		pixel = np.dstack((pixel,bleu))
		pixel_n = pixel[0]
		pixel_n = np.reshape(pixel_n,(32,32,3))

		ens.append(pixel_n)
	for e in ens: 
        	new_ens.append(e[:16,:16])		#en haut  gauche 16 * 16 
       		new_ens.append(e[:16,16:])		#en haut  droite 
        	new_ens.append(e[16:,:16])		#en bas  gauche  
        	new_ens.append(e[16:,16:])		#en bas  droite  
	#return (len(new_ens))
	return array(new_ens)
